// --- Utility Functions --- // ... 前方 parseExcelDate 保持不變 ... const processSheetData = (sheet, existingData) => { const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" }); let { fR, fS, fH, fN } = existingData; EXCEL_ROWS_START.forEach(rS => { EXCEL_COLS_INDEX.forEach(cI => { const dateKey = parseExcelDate(rows[rS]?.[cI]); if (!dateKey) return; // 【微調 1】 初始化時加入 ST 欄位，確保 Index 能找到對應的 Key if (!fR[dateKey]) fR[dateKey] = { GT:{}, BB:{}, GP:{}, ST:{} }; if (!fS[dateKey]) fS[dateKey] = {}; // 【微調 2】 加入 ST 的標題讀取，確保顯示層不會報錯 fH[dateKey] = { GT: (rows[rS-4]?.[cI] || "").toString().trim(), BB: (rows[rS-3]?.[cI] || "").toString().trim(), GP: (rows[rS-2]?.[cI] || "").toString().trim(), ST: (rows[rS-5]?.[cI] || "特殊值班").toString().trim() // 假設在 GT 上方一行 }; fN[dateKey] = (rows[rS-1]?.[cI] || "").toString().trim(); for (let r = rS + 2; r <= rS + 30; r += 2) { const name = (rows[r]?.[1] || "").trim(); if (!STAFF_LIST.includes(name)) continue; const code = (rows[r+1]?.[cI] || "").toString().trim(); if (code === "" || code.match(/休|例|休息|無排班|公假|婚假|喪假|病假|事假|特休|補休/)) { fS[dateKey][name] = "休假"; } else if (code.match(/辦公|支援|居家|其他|其它/)) { fS[dateKey][name] = "辦公"; } else { // 【微調 3】 更新 Regex，加入 ST 的比對 const match = code.match(/(GT|BB|GP|ST)-([A-F])/i); if (match) { const v = match[1].toUpperCase(); // 現在這裡可能是 GT, BB, GP 或 ST const s = ["A", "C", "E"].includes(match[2].toUpperCase()) ? "A" : "B"; fR[dateKey][v][name] = { shift: s }; fS[dateKey][name] = ""; } else { fS[dateKey][name] = "辦公"; } } } }); }); return { fR, fS, fH, fN }; };
